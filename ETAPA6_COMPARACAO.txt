│ ANTES: Leitura de Disco em Cada Requisição                                  │

async function carregarFontes(pdfDoc) {
  const ler = (nome) => {
    try {
      return fs.readFileSync(path.join(process.cwd(), nome));  // ❌ I/O DISCO
    } catch {
      return null;
    }
  };

  return {
    regular:    await embutir(ler(FONT_PATHS.cormorantRegular)),    // ❌ I/O
    bold:       await embutir(ler(FONT_PATHS.cormorantBold)),       // ❌ I/O
    italic:     await embutir(ler(FONT_PATHS.cormorantItalic)),     // ❌ I/O
    boldItalic: await embutir(ler(FONT_PATHS.cormorantBoldItalic)), // ❌ I/O
    sans:       await embutir(ler(FONT_PATHS.openSansRegular)),     // ❌ I/O
    sansBold:   await embutir(ler(FONT_PATHS.openSansBold)),        // ❌ I/O
  };
}

📊 IMPACTO (por requisição):
   • 6 leituras de arquivo (fs.readFileSync)
   • ~2.4 MB transferidos do disco
   • ~10-50ms de latência adicionada
   • Contenção em requisições simultâneas

┌─────────────────────────────────────────────────────────────────────────────┐
│ DEPOIS: Cache em Memória (Pré-carregado)                                    │

// ✅ PRÉ-CARREGADO NO STARTUP (executa UMA ÚNICA VEZ)
const FONT_BYTES_CACHE = (() => {
  const dir = process.cwd();
  const lerFonte = (nome) => {
    try {
      return fs.readFileSync(path.join(dir, nome));  // ✅ Executa UMA VEZ
    } catch {
      return null;
    }
  };

  return {
    cormorantRegular:    lerFonte(FONT_PATHS.cormorantRegular),
    cormorantBold:       lerFonte(FONT_PATHS.cormorantBold),
    cormorantItalic:     lerFonte(FONT_PATHS.cormorantItalic),
    cormorantBoldItalic: lerFonte(FONT_PATHS.cormorantBoldItalic),
    openSansRegular:     lerFonte(FONT_PATHS.openSansRegular),
    openSansBold:        lerFonte(FONT_PATHS.openSansBold),
  };
})();

// ✅ REUTILIZA BYTES JÁ CARREGADOS (nenhum I/O)
async function carregarFontes(pdfDoc) {
  const embutir = async (bytes, fallback) =>
    bytes ? pdfDoc.embedFont(bytes) : pdfDoc.embedFont(fallback);

  return {
    regular:    await embutir(FONT_BYTES_CACHE.cormorantRegular),    // ✅ Memória
    bold:       await embutir(FONT_BYTES_CACHE.cormorantBold),       // ✅ Memória
    italic:     await embutir(FONT_BYTES_CACHE.cormorantItalic),     // ✅ Memória
    boldItalic: await embutir(FONT_BYTES_CACHE.cormorantBoldItalic), // ✅ Memória
    sans:       await embutir(FONT_BYTES_CACHE.openSansRegular),     // ✅ Memória
    sansBold:   await embutir(FONT_BYTES_CACHE.openSansBold),        // ✅ Memória
  };
}

📊 IMPACTO (por requisição):
   • 0 leituras de arquivo
   • 0 bytes transferidos do disco
   • ~0ms de latência adicional
   • Sem contenção (todos acessam memória)
